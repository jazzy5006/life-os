        let state = { totalExercises: 0, completedExercises: 0, checkedIds: [] };

        function renderTimeline(filter) {
            const container = document.getElementById('timeline-container');
            container.innerHTML = ''; 
            scheduleData.forEach((item, idx) => {
                if (filter !== 'all' && !item.categories.includes(filter)) return;
                const node = document.createElement('div');
                node.className = `relative pl-10 py-4 border-l-2 ${idx === 0 ? 'border-transparent' : 'border-stone-200'}`;
                let iconBg = item.categories.includes("medical") ? "bg-rose-100" : (item.categories.includes("puppy") ? "bg-amber-100" : "bg-sky-100");
                node.innerHTML = `<div class="absolute -left-[17px] top-5 w-8 h-8 rounded-full ${iconBg} border-2 border-white flex items-center justify-center text-sm">${item.icon}</div><div class="bg-stone-50 p-3 rounded-lg border border-stone-200">
                    <div class="flex justify-between items-baseline mb-1"><h3 class="font-bold text-stone-800">${item.title}</h3><span class="text-xs font-bold text-stone-500">${item.time}</span></div>
                    <p class="text-sm text-stone-600">${item.desc}</p></div>`;
                container.appendChild(node);
            });
        }

        function renderWorkout() {
            const container = document.getElementById('workout-container');
            container.innerHTML = '';
            state.totalExercises = 0; state.completedExercises = 0;
            const saved = localStorage.getItem('pt_daily_progress');
            state.checkedIds = saved ? JSON.parse(saved) : [];

            workoutData.forEach(section => {
                const wrap = document.createElement('div'); wrap.className = "mb-4";
                wrap.innerHTML = `<h3 class="text-xs font-bold text-stone-400 uppercase tracking-widest mb-3 border-b border-stone-700 pb-1">${section.group}</h3>`;
                const ul = document.createElement('ul'); ul.className = "space-y-3";
                section.exercises.forEach((ex, idx) => {
                    state.totalExercises++;
                    const id = `ex-${section.group.replace(/\s+/g, '-')}-${idx}`;
                    const ch = state.checkedIds.includes(id);
                    if(ch) state.completedExercises++;
                    const li = document.createElement('li'); li.className = "flex items-start space-x-3";
                    li.innerHTML = `<input type="checkbox" id="${id}" class="pt-checkbox" ${ch ? 'checked' : ''} onchange="toggleEx('${id}')">
                        <label for="${id}" class="cursor-pointer select-none">
                            <div class="text-[15px] font-semibold ${ch ? 'completed-text' : 'text-stone-200'}" id="n-${id}">${ex.name}</div>
                            <div class="text-xs ${ch ? 'completed-text' : 'text-stone-500'}" id="d-${id}">${ex.dosage}</div>
                        </label>`;
                    ul.appendChild(li);
                });
                wrap.appendChild(ul); container.appendChild(wrap);
            });
            document.getElementById('total-count').innerText = state.totalExercises;
            updateProgress();
        }

        window.toggleEx = function(id) {
            const cb = document.getElementById(id);
            if(cb.checked) { state.completedExercises++; state.checkedIds.push(id); }
            else { state.completedExercises--; state.checkedIds = state.checkedIds.filter(i => i !== id); }
            localStorage.setItem('pt_daily_progress', JSON.stringify(state.checkedIds));
            renderWorkout();
        }

        window.resetPTProgress = function() {
            if(confirm("Reset PT progress for today?")) { localStorage.removeItem('pt_daily_progress'); renderWorkout(); }
        }

        function updateProgress() {
            const pct = Math.round((state.completedExercises / state.totalExercises) * 100);
            document.getElementById('progress-bar').style.width = `${pct}%`;
            document.getElementById('progress-text').innerText = `${pct}%`;
            document.getElementById('completed-count').innerText = state.completedExercises;
        }

        document.querySelectorAll('.filter-btn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const f = e.target.getAttribute('data-filter');
                document.querySelectorAll('.filter-btn').forEach(b => {
                    b.classList.remove('bg-stone-800', 'text-white', 'bg-amber-500', 'bg-rose-500', 'bg-sky-500');
                    b.classList.add('bg-stone-100', 'text-stone-600');
                });
                e.target.classList.remove('bg-stone-100', 'text-stone-600');
                if(f==='all') e.target.classList.add('bg-stone-800', 'text-white');
                else if(f==='puppy') e.target.classList.add('bg-amber-500', 'text-white');
                else if(f==='medical') e.target.classList.add('bg-rose-500', 'text-white');
                else e.target.classList.add('bg-sky-500', 'text-white');
                renderTimeline(f);
            });
        });
